```py
heap = [1,2,9,5,3,8,4,7]

heapify(heap) # [1,2,4,5,3,9,8,7]
heappush(heap, 6) # [1,2,4,5,3,9,8,7,6]
heappop(heap) # 1, [2,3,4,5,6,9,8,7]
```
